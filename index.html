  95. func writeSource(buf *bytes.Buffer, src []byte, prof *profile.Profile) {

  96.         bi := 0

  97.         si := 0

  98.         line := 1

  99.         col := 1

  100.         cov0 := false

  101.         cov1 := false

  102. 

  103.         buf.WriteString("<ol>")

  104. 

  105.         for si < len(src) {

  106.                 if col == 1 {

  107.                         buf.WriteString("<li>")

  108.                         if cov0 {

  109.                                 buf.WriteString(`<span class="cov0">`)

  110.                         }

  111.                         if cov1 {

  112.                                 buf.WriteString(`<span class="cov1">`)

  113.                         }

  114.                 }

  115. 

  116.                 if len(prof.Blocks) > bi {

  117.                         block := prof.Blocks[bi]

  118.                         if block.StartLine == line && block.StartCol == col {

  119.                                 if block.Count == 0 {

  120.                                         buf.WriteString(`<span class="cov0">`)

  121.                                         cov0 = true

  122.                                 } else {

  123.                                         buf.WriteString(`<span class="cov1">`)

  124.                                         cov1 = true

  125.                                 }

  126.                         }

  127.                         if block.EndLine == line && block.EndCol == col || line > block.EndLine {

  128.                                 buf.WriteString(`</span>`)

  129.                                 bi++

  130.                                 cov0 = false

  131.                                 cov1 = false

  132.                                 continue

  133.                         }

  134.                 }

  135. 

  136.                 b := src[si]

  137.                 buf.WriteString(escape(b))

  138. 

  139.                 if b == '\n' {

  140.                         if cov0 || cov1 {

  141.                                 buf.WriteString("</span>")

  142.                         }

  143.                         buf.WriteString("</li>")

  144.                         line++

  145.                         col = 0

  146.                 }

  147. 

  148.                 si++

  149.                 col++

  150.         }

  151. 

  152.         buf.WriteString("</ol>")

  153. }

  18. func Create(profiles []profile.Profile) []Node {

  20.         for _, p := range profiles {

  21.                 addNode(&nodes, strings.Split(p.FileName, "/"), &p)

  27. func addNode(nodes *[]Node, paths []string, p *profile.Profile) {

  28.         name := paths[0]

  29.         nextPaths := paths[1:]

  30. 

  31.         idx := index(*nodes, name)

  32.         if idx == -1 {

  33.                 *nodes = append(*nodes, Node{

  34.                         Name:  name,

  35.                         Files: make([]profile.Profile, 0),

  36.                         Dirs:  make([]Node, 0),

  37.                 })

  38.                 idx = len(*nodes) - 1

  39.         }

  40. 

  41.         n := *nodes

  42.         if len(nextPaths) == 1 {

  43.                 n[idx].Files = append(n[idx].Files, *p)

  44.                 return

  45.         }

  46. 

  47.         addNode(&n[idx].Dirs, nextPaths, p)

  48. }

  50. func index(nodes []Node, name string) int {

  51.         for i, t := range nodes {

  52.                 if t.Name == name {

  53.                         return i

  54.                 }

  55.         }

  56.         return -1

  57. }

  63.                 if len(t.Files) == 0 && len(t.Dirs) == 1 {

  28. // Coverage returns covered ratio for file

  29. func (prof *Profile) Coverage() float64 {

  30.         var total, covered int64

  31.         for _, b := range prof.Blocks {

  32.                 total += int64(b.NumState)

  33.                 if b.Count > 0 {

  34.                         covered += int64(b.NumState)

  35.                 }

  36.         }

  37. 

  38.         if total == 0 {

  39.                 return 0

  40.         }

  41. 

  42.         return math.Round((float64(covered)/float64(total)*100)*10) / 10

  43. }

  19. var reg = regexp.MustCompile(`^(.+):([0-9]+).([0-9]+),([0-9]+).([0-9]+) ([0-9]+) ([0-9]+)$`)

  21. // Read is reads profiling data

  22. func Read(r io.Reader) ([]Profile, error) {

  23.         files := make(map[string]*Profile)

  24.         modeOk := false

  25.         id := 1

  26. 

  27.         s := bufio.NewScanner(r)

  28.         for s.Scan() {

  29.                 line := s.Text()

  30. 

  31.                 // first line must be "mode: xxx"

  32.                 if !modeOk {

  33.                         const p = "mode: "

  34.                         if !strings.HasPrefix(line, p) || line == p {

  35.                                 return nil, fmt.Errorf("first line must be mode: %q", line)

  36.                         }

  37.                         modeOk = true

  38.                         continue

  39.                 }

  40. 

  41.                 m := reg.FindStringSubmatch(line)

  42.                 if m == nil {

  43.                         return nil, fmt.Errorf("%q does not match expected format: %v", line, reg)

  44.                 }

  45. 

  46.                 fileName := m[1]

  47.                 p := files[fileName]

  48. 

  49.                 if p == nil {

  50.                         p = &Profile{ID: id, FileName: fileName}

  51.                         files[fileName] = p

  52.                         id++

  53.                 }

  54. 

  55.                 p.Blocks = append(p.Blocks, Block{

  56.                         StartLine: toInt(m[2]),

  57.                         StartCol:  toInt(m[3]),

  58.                         EndLine:   toInt(m[4]),

  59.                         EndCol:    toInt(m[5]),

  60.                         NumState:  toInt(m[6]),

  61.                         Count:     toInt(m[7]),

  62.                 })

  63.         }

  64. 

  65.         return toProfiles(files)

  66. }

  76. func toProfiles(files map[string]*Profile) ([]Profile, error) {

  77.         dirs, err := importPath(files)

  78.         if err != nil {

  79.                 return nil, err

  80.         }

  81. 

  82.         profiles := make([]Profile, 0, len(files))

  83.         for _, p := range files {

  84.                 p.Blocks = filterBlocks(p.Blocks)

  85.                 sort.SliceStable(p.Blocks, func(i, j int) bool {

  86.                         bi, bj := p.Blocks[i], p.Blocks[j]

  87.                         return bi.StartLine < bj.StartLine || bi.StartLine == bj.StartLine && bi.StartCol < bj.StartCol

  88.                 })

  89. 

  90.                 p.Dir = dirs[path.Dir(p.FileName)]

  91. 

  92.                 profiles = append(profiles, *p)

  93.         }

  94. 

  95.         sort.SliceStable(profiles, func(i, j int) bool {

  96.                 return profiles[i].FileName < profiles[j].FileName

  97.         })

  98. 

  99.         return profiles, nil

  100. }

  102. func filterBlocks(blocks []Block) []Block {

  103.         index := func(bs []Block, b *Block) int {

  104.                 for i, bb := range bs {

  105.                         if bb.StartLine == b.StartLine &&

  106.                                 bb.StartCol == b.StartCol &&

  107.                                 bb.EndLine == b.EndLine &&

  108.                                 bb.EndCol == b.EndCol {

  109.                                 return i

  110.                         }

  111.                 }

  112.                 return -1

  113.         }

  114. 

  115.         newBlocks := make([]Block, 0, len(blocks))

  116.         for _, b := range blocks {

  117.                 i := index(newBlocks, &b)

  118.                 if i == -1 {

  119.                         newBlocks = append(newBlocks, b)

  120.                         continue

  121.                 }

  122.                 if b.Count > 0 {

  123.                         newBlocks[i] = b

  124.                 }

  125.         }

  126. 

  127.         return newBlocks

  128. }

  130. func importPath(files map[string]*Profile) (map[string]string, error) {

  131.         dirs := make([]string, 0, len(files))

  132.         pkgs := make(map[string]string)

  133. 

  134.         for _, p := range files {

  135.                 if p.IsRelativeOrAbsolute() {

  136.                         continue

  137.                 }

  138. 

  139.                 dir := path.Dir(p.FileName)

  140.                 if _, ok := pkgs[dir]; !ok {

  141.                         pkgs[dir] = ""

  142.                         dirs = append(dirs, dir)

  143.                 }

  144.         }

  145. 

  146.         if len(dirs) == 0 {

  147.                 return pkgs, nil

  148.         }

  149. 

  150.         cmdName := filepath.Join(runtime.GOROOT(), "bin/go")

  151.         args := append([]string{"list", "-e", "-json"}, dirs...)

  152.         cmd := exec.Command(cmdName, args...)

  153. 

  154.         stdout, err := cmd.Output()

  155.         if err != nil {

  156.                 return nil, err

  157.         }

  158. 

  159.         type pkg struct {

  160.                 Dir        string

  161.                 ImportPath string

  162.                 Error      *struct {

  163.                         Err string

  164.                 }

  165.         }

  166. 

  167.         dec := json.NewDecoder(bytes.NewReader(stdout))

  168.         for {

  169.                 var p pkg

  170.                 err := dec.Decode(&p)

  171.                 if err == io.EOF {

  172.                         break

  173.                 }

  174.                 if err != nil {

  175.                         return nil, fmt.Errorf("decoding go list json: %v", err)

  176.                 }

  177.                 if p.Error != nil {

  178.                         return nil, fmt.Errorf(p.Error.Err)

  179.                 }

  180.                 pkgs[p.ImportPath] = p.Dir

  181.         }

  182. 

  183.         return pkgs, nil

  184. }
