  24. func ReadProfile(r io.Reader) ([]profile.Profile, error) {

  25.         files := make(map[string]*profile.Profile)

  26.         modeOk := false

  27.         id := 1

  28. 

  29.         s := bufio.NewScanner(r)

  30.         for s.Scan() {

  31.                 line := s.Text()

  32. 

  33.                 // first line must be "mode: xxx"

  34.                 if !modeOk {

  35.                         const p = "mode: "

  36.                         if !strings.HasPrefix(line, p) || line == p {

  37.                                 return nil, fmt.Errorf("first line must be mode: %q", line)

  38.                         }

  39.                         modeOk = true

  40.                         continue

  41.                 }

  42. 

  43.                 matches := reg.FindStringSubmatch(line)

  44.                 if matches == nil {

  45.                         return nil, fmt.Errorf("%q does not match expected format: %v", line, reg)

  46.                 }

  47. 

  48.                 fileName := matches[1]

  49.                 p := files[fileName]

  50. 

  51.                 if p == nil {

  52.                         p = &profile.Profile{ID: id, FileName: fileName}

  53.                         files[fileName] = p

  54.                         id++

  55.                 }

  56. 

  57.                 p.Blocks = append(p.Blocks, profile.Block{

  58.                         StartLine: toInt(matches[2]),

  59.                         StartCol:  toInt(matches[3]),

  60.                         EndLine:   toInt(matches[4]),

  61.                         EndCol:    toInt(matches[5]),

  62.                         NumState:  toInt(matches[6]),

  63.                         Count:     toInt(matches[7]),

  64.                 })

  65.         }

  66. 

  67.         return toProfiles(files)

  68. }

  78. func toProfiles(files map[string]*profile.Profile) ([]profile.Profile, error) {

  79.         dirs, err := makeImportDirMap(files)

  80.         if err != nil {

  81.                 return nil, err

  82.         }

  83. 

  84.         profiles := make([]profile.Profile, 0, len(files))

  85.         for _, p := range files {

  86.                 p.Blocks = filterBlocks(p.Blocks)

  87.                 sort.SliceStable(p.Blocks, func(i, j int) bool {

  88.                         bi, bj := p.Blocks[i], p.Blocks[j]

  89.                         return bi.StartLine < bj.StartLine || bi.StartLine == bj.StartLine && bi.StartCol < bj.StartCol

  90.                 })

  91. 

  92.                 p.Dir = dirs[path.Dir(p.FileName)]

  93. 

  94.                 profiles = append(profiles, *p)

  95.         }

  96. 

  97.         sort.SliceStable(profiles, func(i, j int) bool {

  98.                 return profiles[i].FileName < profiles[j].FileName

  99.         })

  100. 

  101.         return profiles, nil

  102. }

  104. func filterBlocks(blocks []profile.Block) []profile.Block {

  105.         index := func(bs []profile.Block, b *profile.Block) int {

  106.                 for i, bb := range bs {

  107.                         if bb.StartLine == b.StartLine &&

  108.                                 bb.StartCol == b.StartCol &&

  109.                                 bb.EndLine == b.EndLine &&

  110.                                 bb.EndCol == b.EndCol {

  111.                                 return i

  112.                         }

  113.                 }

  114.                 return -1

  115.         }

  116. 

  117.         newBlocks := make([]profile.Block, 0, len(blocks))

  118.         for _, b := range blocks {

  119.                 i := index(newBlocks, &b)

  120.                 if i == -1 {

  121.                         newBlocks = append(newBlocks, b)

  122.                         continue

  123.                 }

  124.                 if b.Count > 0 {

  125.                         newBlocks[i] = b

  126.                 }

  127.         }

  128. 

  129.         return newBlocks

  130. }

  132. func makeImportDirMap(files map[string]*profile.Profile) (map[string]string, error) {

  133.         stdout, err := execGoList(files)

  134.         if err != nil {

  135.                 return nil, err

  136.         }

  137. 

  138.         pkgs := make(map[string]string)

  139.         if len(stdout) == 0 {

  140.                 return pkgs, nil

  141.         }

  142. 

  143.         type pkg struct {

  144.                 Dir        string

  145.                 ImportPath string

  146.                 Error      *struct {

  147.                         Err string

  148.                 }

  149.         }

  150. 

  151.         dec := json.NewDecoder(bytes.NewReader(stdout))

  152.         for {

  153.                 var p pkg

  154.                 err := dec.Decode(&p)

  155.                 if err == io.EOF {

  156.                         break

  157.                 }

  158.                 if err != nil {

  159.                         return nil, fmt.Errorf("decoding go list json: %v", err)

  160.                 }

  161.                 if p.Error != nil {

  162.                         return nil, fmt.Errorf(p.Error.Err)

  163.                 }

  164.                 // should have the same result for "pkg.ImportPath" and "path.Dir(Profile.FileName)"

  165.                 pkgs[p.ImportPath] = p.Dir

  166.         }

  167. 

  168.         return pkgs, nil

  169. }

  171. // execute "go list" command

  172. func execGoList(files map[string]*profile.Profile) ([]byte, error) {

  173.         dirs := make([]string, 0, len(files))

  174.         m := make(map[string]struct{})

  175. 

  176.         for _, p := range files {

  177.                 if p.IsRelativeOrAbsolute() {

  178.                         continue

  179.                 }

  180.                 dir := path.Dir(p.FileName)

  181.                 if _, ok := m[dir]; !ok {

  182.                         m[dir] = struct{}{}

  183.                         dirs = append(dirs, dir)

  184.                 }

  185.         }

  186. 

  187.         if len(dirs) == 0 {

  188.                 return make([]byte, 0), nil

  189.         }

  190. 

  191.         cmdName := filepath.Join(runtime.GOROOT(), "bin/go")

  192.         args := append([]string{"list", "-e", "-json"}, dirs...)

  193.         cmd := exec.Command(cmdName, args...)

  194. 

  195.         return cmd.Output()

  196. }

  98. func writeSource(buf *bytes.Buffer, src []byte, prof *profile.Profile) {

  99.         bi := 0

  100.         si := 0

  101.         line := 1

  102.         col := 1

  103.         cov0 := false

  104.         cov1 := false

  105. 

  106.         buf.WriteString("<ol>")

  107. 

  108.         for si < len(src) {

  109.                 if col == 1 {

  110.                         buf.WriteString("<li>")

  111.                         if cov0 {

  112.                                 buf.WriteString(`<span class="cov0">`)

  113.                         }

  114.                         if cov1 {

  115.                                 buf.WriteString(`<span class="cov1">`)

  116.                         }

  117.                 }

  118. 

  119.                 if len(prof.Blocks) > bi {

  120.                         block := prof.Blocks[bi]

  121.                         if block.StartLine == line && block.StartCol == col {

  122.                                 if block.Count == 0 {

  123.                                         buf.WriteString(`<span class="cov0">`)

  124.                                         cov0 = true

  125.                                 } else {

  126.                                         buf.WriteString(`<span class="cov1">`)

  127.                                         cov1 = true

  128.                                 }

  129.                         }

  130.                         if block.EndLine == line && block.EndCol == col || line > block.EndLine {

  131.                                 buf.WriteString(`</span>`)

  132.                                 bi++

  133.                                 cov0 = false

  134.                                 cov1 = false

  135.                                 continue

  136.                         }

  137.                 }

  138. 

  139.                 b := src[si]

  140.                 buf.WriteString(escape(b))

  141. 

  142.                 if b == '\n' {

  143.                         if cov0 || cov1 {

  144.                                 buf.WriteString("</span>")

  145.                         }

  146.                         buf.WriteString("</li>")

  147.                         line++

  148.                         col = 0

  149.                 }

  150. 

  151.                 si++

  152.                 col++

  153.         }

  154. 

  155.         buf.WriteString("</ol>")

  156. }

  10. // Node is single node of directory tree

  11. type Node struct {

  12.         Name  string

  13.         Files []profile.Profile

  14.         Dirs  []Node

  17. // Create returns directory tree

  18. func Create(profiles []profile.Profile) []Node {

  19.         nodes := make([]Node, 0)

  20.         for _, p := range profiles {

  21.                 addNode(&nodes, strings.Split(p.FileName, "/"), &p)

  22.         }

  23. 

  24.         return mergeSingreDir(nodes)

  25. }

  26. 

  27. func addNode(nodes *[]Node, paths []string, p *profile.Profile) {

  28.         name := paths[0]

  29.         nextPaths := paths[1:]

  30. 

  31.         idx := index(*nodes, name)

  32.         if idx == -1 {

  33.                 *nodes = append(*nodes, Node{Name: name})

  34.                 idx = len(*nodes) - 1

  35.         }

  36. 

  37.         n := *nodes

  38.         if len(nextPaths) == 1 {

  39.                 n[idx].Files = append(n[idx].Files, *p)

  40.                 return

  41.         }

  42. 

  43.         addNode(&n[idx].Dirs, nextPaths, p)

  44. }

  45. 

  46. func index(nodes []Node, name string) int {

  47.         for i, t := range nodes {

  48.                 if t.Name == name {

  49.                         return i

  50.                 }

  51.         }

  52.         return -1

  53. }

  54. 

  55. // merge directories with no files and only one child directory

  56. //

  57. // path/

  58. //   to/

  59. //     file.go

  60. //

  61. // to

  62. //

  63. // path/to/

  64. //   file.go

  65. //

  66. func mergeSingreDir(nodes []Node) []Node {

  67.         for i, n := range nodes {

  68.                 if len(n.Dirs) == 0 {

  69.                         continue

  70.                 }

  71. 

  72.                 mergeSingreDir(n.Dirs)

  73.                 if len(n.Files) > 0 || len(n.Dirs) != 1 {

  74.                         continue

  75.                 }

  76. 

  77.                 sub := n.Dirs[0]

  78.                 nodes[i].Name = filepath.Join(n.Name, sub.Name)

  79.                 nodes[i].Files = sub.Files

  80.                 nodes[i].Dirs = sub.Dirs

  81.         }

  82.         return nodes

  83. }

  28. // Coverage returns covered ratio for file

  29. func (prof *Profile) Coverage() float64 {

  30.         var total, covered int64

  31.         for _, b := range prof.Blocks {

  32.                 total += int64(b.NumState)

  33.                 if b.Count > 0 {

  34.                         covered += int64(b.NumState)

  35.                 }

  36.         }

  37. 

  38.         if total == 0 {

  39.                 return 0

  40.         }

  41. 

  42.         return math.Round((float64(covered)/float64(total)*100)*10) / 10

  43. }
