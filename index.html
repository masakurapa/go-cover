  124.                                         buf.WriteString(`<span class="cov1">`)

  125.                                         cov1 = true

  126.                                 }

  127.                         }

  128.                         if block.EndLine == line && block.EndCol == col || line > block.EndLine {

  132.                                 cov1 = false

  134.                         }

  142.                                 buf.WriteString("</span>")

  146.                         col = 0

  151.         }

  54.                 }

  55.         }
